# M22-PRD: Planificador de Catering y MRP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the Catering Planning and Material Requirements Planning (MRP) engine to recursively explode recipes, calculate required production orders, and generate a dynamic purchase list based on warehouse stock.

**Architecture:** 
1. Backend schemas for catering requests and MRP results.
2. Backend REST endpoints for catering CRUD.
3. Backend recursive MRP engine (`/plan`) that builds a BOM tree and consolidates raw material deficits.
4. Backend bulk order generator (`/generate-orders`).
5. Frontend interfaces for managing events and a 3-column MRP console with PDF export capabilities.

**Tech Stack:** Python (FastAPI, Pytest), Supabase (PostgreSQL), TypeScript (Next.js, Tailwind CSS, `@react-pdf/renderer`).

```python
# In backend/main.py
@app.post("/production/catering/{req_id}/plan", response_model=MRPResultResponse, tags=["Production"])
async def generate_mrp_plan(req_id: str, body: MRPPlanRequest, org_id: str = Depends(get_active_org_id), db=Depends(get_db), _=Depends(require_permission("production.manage_catering"))):
    lines = db.table("catering_request_lines").select("item_id, qty_base").eq("request_id", req_id).execute().data or []
    if not lines: raise HTTPException(400, "Empty request")
    
    # Pre-fetch all recipes and ingredients to memory for fast tree traversal
    all_recipes = db.table("recipes").select("*").eq("org_id", org_id).execute().data or []
    recipe_map = {r["item_id"]: r for r in all_recipes}
    all_ingredients = db.table("recipe_ingredients").select("*, items(name, uom_base(name))").execute().data or []
    ing_map = {}
    for i in all_ingredients:
        ing_map.setdefault(i["recipe_id"], []).append(i)
        
    production_needs = {} # item_id -> total qty to produce
    raw_materials = {} # item_id -> total qty needed
    item_names = {}
    uom_names = {}
    
    # 1. Recursive explosion function
    def explode(item_id, qty):
        recipe = recipe_map.get(item_id)
        if recipe:
            # It's a manufactured item
            yield_base = Decimal(str(recipe["yield_qty_base"]))
            scale = Decimal(str(qty)) / yield_base if yield_base > 0 else Decimal("0")
            production_needs[item_id] = production_needs.get(item_id, Decimal("0")) + Decimal(str(qty))
            
            for ing in ing_map.get(recipe["id"], []):
                ing_id = ing["item_id"]
                item_names[ing_id] = ing.get("items", {}).get("name", "Unknown")
                uom_names[ing_id] = ing.get("items", {}).get("uom_base", {}).get("name", "")
                ing_qty = Decimal(str(ing["qty_base"])) * scale
                explode(ing_id, ing_qty)
        else:
            # It's a raw material
            raw_materials[item_id] = raw_materials.get(item_id, Decimal("0")) + Decimal(str(qty))

    # Seed the names for root items
    item_ids = [l["item_id"] for l in lines]
    items_meta = db.table("items").select("id, name, uom_base(name)").in_("id", item_ids).execute().data or []
    for im in items_meta:
        item_names[im["id"]] = im["name"]
        uom_names[im["id"]] = im.get("uom_base", {}).get("name", "")

    # Run explosion
    for line in lines:
        explode(line["item_id"], line["qty_base"])
        
    # 2. Build Production Plan
    prod_plan = []
    for pid, qty in production_needs.items():
        rec = recipe_map[pid]
        prod_plan.append({
            "item_id": pid, "item_name": item_names.get(pid, ""), "uom_name": uom_names.get(pid, ""),
            "qty_to_produce": float(qty), "recipe_id": rec["id"]
        })
        
    # 3. Analyze Stock for Purchase List
    purch_list = []
    if raw_materials:
        raw_ids = list(raw_materials.keys())
        stock_res = db.table("stock").select("item_id, qty_base, qty_reserved").eq("warehouse_id", str(body.warehouse_id)).in_("item_id", raw_ids).execute().data or []
        stock_map = {s["item_id"]: Decimal(str(s["qty_base"])) - Decimal(str(s["qty_reserved"])) for s in stock_res}
        
        for rid, needed in raw_materials.items():
            avail = stock_map.get(rid, Decimal("0"))
            deficit = max(Decimal("0"), needed - avail)
            purch_list.append({
                "item_id": rid, "item_name": item_names.get(rid, ""), "uom_name": uom_names.get(rid, ""),
                "qty_needed": float(needed), "qty_available": float(avail), "qty_deficit": float(deficit)
            })
            
    return {"production_plan": prod_plan, "purchase_list": purch_list}
```

```python
@app.post("/production/catering/{req_id}/generate-orders", tags=["Production"])
async def generate_catering_orders(req_id: str, body: GenerateOrdersRequest, org_id: str = Depends(get_active_org_id), user=Depends(get_current_user), db=Depends(get_db), _=Depends(require_permission("production.create"))):
    # 1. Run explosion to get the plan
    plan_data = await generate_mrp_plan(req_id, MRPPlanRequest(warehouse_id=body.warehouse_id), org_id, db)
    prod_plan = plan_data.get("production_plan", [])
    
    if not prod_plan:
        raise HTTPException(400, "No production required for this event")
        
    # 2. Get sequence number for today
    today_date = datetime.now(CARACAS_TZ).strftime("%Y-%m-%d")
    today_str = datetime.now(CARACAS_TZ).strftime("%Y%m%d")
    count_res = db.table("production_orders").select("id", count="exact").gte("created_at", today_date).execute()
    seq = (count_res.count or 0) + 1
    
    generated_orders = []
    
    # 3. Create orders iteratively
    for item in prod_plan:
        order_number = f"OP-{today_str}-{seq:04d}"
        seq += 1
        
        order_data = {
            "org_id": org_id, "order_number": order_number, "item_id": item["item_id"],
            "recipe_id": item["recipe_id"], "warehouse_id": str(body.warehouse_id),
            "target_warehouse_id": str(body.target_warehouse_id),
            "qty_ordered_base": item["qty_to_produce"], "scheduled_date": body.scheduled_date,
            "status": "pending", "priority": "high", "catering_request_id": req_id,
            "created_by": user.id
        }
        res = db.table("production_orders").insert(order_data).execute()
        order_id = res.data[0]["id"]
        generated_orders.append(res.data[0])
        
        # Save planned consumptions
        recipe_yield_base = Decimal(str(db.table("recipes").select("yield_qty_base").eq("id", item["recipe_id"]).execute().data[0]["yield_qty_base"]))
        scale_factor = Decimal(str(item["qty_to_produce"])) / recipe_yield_base if recipe_yield_base > 0 else Decimal("0")
        
        ing_res = db.table("recipe_ingredients").select("item_id, qty_base").eq("recipe_id", item["recipe_id"]).execute()
        for ing in (ing_res.data or []):
            planned_qty = Decimal(str(ing["qty_base"])) * scale_factor
            db.table("production_order_consumptions").insert({
                "order_id": order_id, "item_id": ing["item_id"], "qty_planned_base": float(planned_qty)
            }).execute()
            
    # Mark request as confirmed
    db.table("catering_requests").update({"status": "confirmed"}).eq("id", req_id).execute()
            
    return {"ok": True, "generated": len(generated_orders), "orders": generated_orders}
```

```tsx
                                    <span className="font-mono font-black text-primary bg-primary/5 px-2 py-0.5 rounded">{l.qty_base} <span className="text-[10px] text-text-secondary uppercase ml-0.5">{l.items?.uom_base?.name}</span></span>
                                </li>
                            ))}
                        </ul>
                    </div>

                    <div className="bg-surface p-6 rounded-2xl border border-border shadow-sm">
                        <div className="flex justify-between items-center mb-4">
                            <h2 className="text-xs font-black text-text-secondary uppercase tracking-widest">2. Plan de Producción</h2>
                            {request.status === 'planning' && (
                                <button onClick={handleGenerateOrders} disabled={generating} className="text-[10px] font-black bg-primary text-white px-2 py-1 rounded-lg flex items-center gap-1 uppercase tracking-tighter transition-all hover:scale-105 active:scale-95 shadow-md shadow-primary/10">
                                    <Factory className="w-3 h-3"/> Generar OPs
                                </button>
                            )}
                        </div>
                        <ul className="space-y-3">
                            {plan.production_plan.map((p: any, i: number) => (
                                <li key={i} className="flex justify-between text-sm items-center">
                                    <span className="font-medium text-text-primary">{p.item_name}</span>
                                    <span className="font-mono font-black text-primary">{p.qty_to_produce} <span className="text-[10px] text-text-secondary uppercase ml-0.5">{p.uom_name}</span></span>
                                </li>
                            ))}
                        </ul>
                    </div>

                    <div className="bg-surface p-6 rounded-2xl border border-error/20 shadow-sm bg-error/[0.02]">
                        <div className="flex justify-between items-center mb-4">
                            <h2 className="text-xs font-black text-error uppercase tracking-widest">3. Lista de Compras</h2>
                            <PDFDownloadLink document={<MRPPurchaseListPDF eventName={request.name} items={plan.purchase_list.filter((i:any) => i.qty_deficit > 0)} />} fileName={`compras_${request.name}.pdf`}>
                                <button className="text-[10px] font-black bg-surface-raised border border-border px-2 py-1 rounded-lg flex items-center gap-1 uppercase tracking-tighter hover:bg-surface transition-colors shadow-sm">
                                    <FileText className="w-3 h-3"/> Exportar PDF
                                </button>
                            </PDFDownloadLink>
                        </div>
                        <ul className="space-y-3">
                            {plan.purchase_list.filter((i:any) => i.qty_deficit > 0).map((p: any, i: number) => (
                                <li key={i} className="flex justify-between text-sm border-b border-border/50 pb-2 items-center last:border-0">
                                    <span className="font-medium text-text-primary">{p.item_name}</span>
                                    <span className="font-mono font-black text-error">-{p.qty_deficit} <span className="text-[10px] opacity-60 uppercase ml-0.5">{p.uom_name}</span></span>
                                </li>
                            ))}
                            {plan.purchase_list.filter((i:any) => i.qty_deficit > 0).length === 0 && (
                                <li className="text-center py-10">
                                    <p className="text-sm font-bold text-success uppercase">Stock Suficiente</p>
                                    <p className="text-[10px] text-text-secondary mt-1">No se requieren compras para este plan.</p>
                                </li>
                            )}
                        </ul>
                    </div>
                </div>
            )}
        </div>
    )
}
```
